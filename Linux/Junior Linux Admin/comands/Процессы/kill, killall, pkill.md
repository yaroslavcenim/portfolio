Процессам можно отправлять сигналы. Самые важные:
1 HUP перезагрузить конфигурацию
9 KILL принудительно убить (нельзя игнорировать)
15 TERM завершить (по умолчанию)
19 STOP приостановить
18 CONT продолжить после STOP

kill — отправить сигнал по PID
  kill 1234                # сигнал 15 (TERM)
  kill -9 1234             # сигнал 9 (KILL)
  kill -HUP 1234           # перезагрузить конфиг

killall — убить все процессы по имени
  killall nginx            # завершить все nginx
  killall -9 sshd          # принудительно убить все sshd

pkill — убить по имени и другим критериям
  pkill -u root            # убить все процессы root
  pkill -f "python script" # убить по полной команде
  pkill -9 -u alice        # убить все процессы alice
